CELL_SIZE) // CELL_SIZE) * CELL_SIZE y = random.randint(0, (HEIGHT - CELL_SIZE) // CELL_SIZE) * CELL_SIZE if [x, y] not in snake_body: return (x, y) def game_loop(): # Initial snake setup snake_x = WIDTH // 2 snake_y = HEIGHT // 2 snake_body = [[snake_x, snake_y]] snake_length = 1 # Initial movement dx = 0 dy = 0 # Generate first food food_x, food_y = generate_food(snake_body) while True: # Event handling for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() elif event.type == pygame.KEYDOWN: # Move the snake with arrow keys or WASD if event.key in (pygame.K_UP, pygame.K_w) and dy == 0: dx, dy = 0, -CELL_SIZE elif event.key in (pygame.K_DOWN, pygame.K_s) and dy == 0: dx, dy = 0, CELL_SIZE elif event.key in (pygame.K_LEFT, pygame.K_a) and dx == 0: dx, dy = -CELL_SIZE, 0 elif event.key in (pygame.K_RIGHT, pygame.K_d) and dx == 0: dx, dy = CELL_SIZE, 0 # Move snake head snake_x += dx snake_y += dy snake_body.insert(0, [snake_x, snake_y]) # Check if snake eats food if snake_x == food_x and snake_y == food_y: snake_length += 1 food_x, food_y = generate_food(snake_body) else: # If no food eaten, pop the tail snake_body.pop() # Check for collision with walls if (snake_x < 0 or snake_x >= WIDTH or snake_y < 0 or snake_y >= HEIGHT): break # Game over # Check for collision with self if [snake_x, snake_y] in snake_body[1:]: break # Game over # Drawing screen.fill(BLACK) draw_snake(snake_body) pygame.draw.rect(screen, RED, (food_x, food_y, CELL_SIZE, CELL_SIZE)) pygame.display.update() clock.tick(10) # Control speed (frames per second) # Game over message screen.fill(BLACK) game_over_text = font.render("Game Over! Press any key to exit.", True, WHITE) screen.blit(game_over_text, (WIDTH // 2 - 180, HEIGHT // 2 - 20)) pygame.display.update() # Wait for a key press to exit while True: for event in pygame.event.get(): if event.type == pygame.KEYDOWN or event.type == pygame.QUIT: pygame.quit() sys.exit() def main(): game_loop() if __name__ == "__main__": main() -------------------------------------------------------------------------------- How to run: 1. Install Pygame if you don’t have it: pip install pygame 2. Save the code above into a file named, for example, snake_game.py. 3.
